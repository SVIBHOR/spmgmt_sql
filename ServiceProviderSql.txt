CREATE TABLE Service_Providers( User_ID VARCHAR2(10) PRIMARY KEY,
 Company_Name VARCHAR2(30) NOT NULL,
 Company_Address VARCHAR2(200) NOT NULL,
 Gst_in VARCHAR2(16) NOT NULL UNIQUE,
PAN VARCHAR2(10) NOT NULL UNIQUE,
Mobile_Number NUMBER(10) NOT NULL UNIQUE,
Bank_Name VARCHAR2(30) NOT NULL,
Account_Number VARCHAR2(16) NOT NULL UNIQUE,
Password VARCHAR2(8) NOT NULL,
Status VARCHAR2(16) DEFAULT 'PENDING',
SPI NUMBER(10) UNIQUE,
Request_Date DATETIME NOT NULL,
Forgot_Pass_Question VARCHAR2(150) NOT NULL,
Fogot_Pass_Answer VARCHAR2(50) NOT NULL);
